def save_to_file(fileline="") :
    if os.name == 'posix':      # RPI linux
        RAMDISKpath = '/run/shm/'
    else :
        RAMDISKpath = 'C:/tmp/'
        
    sfiledata   = RAMDISKpath+"text.txt"
        
    fdat = open(sfiledata,"w")
    fdat.write(fileline+"\n")
    fdat.close()
